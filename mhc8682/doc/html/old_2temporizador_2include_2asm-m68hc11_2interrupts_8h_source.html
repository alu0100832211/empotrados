178 extern inline void
179 set_interrupt_handler (interrupt_vector_id id, interrupt_t handler)
180 {
181  unsigned char *p;
182 
183  /* Set interrupt handler for bootstrap mode.
184  Compute address of that vector (3-bytes for each).
185  Cast to unsigned short to avoid a Gcc warnings if sizeof int == 32. */
186  p = (unsigned char*) ((unsigned short) (0x100 - 3 * (RESET_VECTOR - id)));
187  *p++ = M68HC11_JMP;
188  ((interrupt_t*)p)[0] = handler;
189 
190  /* Set interrupt handler for normal mode. */
191  _vectors_addr[id] = handler;
192 }
