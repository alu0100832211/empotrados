62 extern inline void serial_init (void)

64  ((unsigned volatile short*) &_io_ports[M6812_SCBD])[0] = M6812_DEF_BAUD;

112 extern inline void serial_send (char c)
113 {

129 extern inline unsigned char serial_recv (void)
